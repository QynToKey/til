```ruby
# db/migrate/20260419032335_create_reading_sessions.rb
class CreateReadingSessions < ActiveRecord::Migration[8.1]
  def change
    create_table :reading_sessions do |t|
      t.references :text, null: false, foreign_key: true
      # セッション作成者を users テーブルの id として参照するため、foreign_key オプションで to_table を指定
      t.references :created_by, null: false, foreign_key: { to_table: :users }
      t.string :name
      t.string :invite_token, null: false

      t.timestamps
    end

    # invite_token にユニークインデックスを追加して、重複を防止

  end
end
```

👉 *`invite_token` に `null: false` を追加*
👉 *`invite_token` に `UNIQUE index` を追加*

## 2️⃣ `membership` テーブル作成マイグレーション


```ruby
# db/migrate/20260419040404_create_memberships.rb
class CreateMemberships < ActiveRecord::Migration[8.1]
  def change
    create_table :memberships do |t|
      t.references :user, null: false, foreign_key: true
      t.references :reading_session, null: false, foreign_key: true
      t.integer :session_role, null: false, default: 0 # 0: member, 1: session_admin
      t.integer :mode, null: false, default: 0 # 0: solo, 1: 招待済み, 2: co_reading
      t.timestamps
    end

    add_index :memberships, [:user_id, :reading_session_id], unique: true # 同じユーザーが同じセッションに複数回参加するのを防止
  end
end
```

👉 *`session_role` と `mode` は `enum` 用に `integer` で定義（Rails の `enum` はデフォルト値必須）*
👉 *`user_id` + `reading_session_id` の「複合 UNIQUE index」で同一セッションへの重複参加を防ぐ*
👉 *`mode` は #70 の設計メモにはないが、ER 図上 `memberships.mode` が必要なため含めた*


📝 `force: :cascade` ：
`schema.rb` に自動出力される記述で、`db:schema:load`（スキーマを一から作り直す際）に使われる。

- `force` ： 同名のテーブルが既に存在する場合、一度削除してから作り直す
- `:cascade` ：削除する際に、そのテーブルに依存している他のテーブルも含めて連鎖的に削除する

👉 *ただし、これは開発環境でスキーマを再構築するときだけの話で、通常の `db:migrate` では `force: :cascade` は関係しない。（Rails が自動生成するだけ）*

```ruby
# app/models/reading_session.rb
class ReadingSession < ApplicationRecord
  # リーディングセッションは特定のテキストに対して作成され、ユーザーが参加できるセッションを表すモデル
  belongs_to :text
  belongs_to :creator, class_name: "User", foreign_key: :created_by_id
  has_many :memberships, dependent: :destroy
  has_many :users, through: :memberships

  validates :invite_token, presence: true, uniqueness: true
  validates :text, presence: true
  validates :creator, presence: true
end
```

```ruby
# app/models/membership.rb
class Membership < ApplicationRecord
  # Membership はユーザーがリーディングセッションに参加するためのモデルで、ユーザーとリーディングセッションの多対多の関係を管理する
  belongs_to :user
  belongs_to :reading_session

  enum :session_role, { member: 0, session_admin: 1 }
  enum :mode, { solo: 0, co_reading: 1 }

  validates :user, presence: true
  validates :reading_session, presence: true
  validates :user_id, uniqueness: { scope: :reading_session_id }
end
```

## 5️⃣ セッション作成者に `session_admin` を自動付与

`ReadingSession` モデルに以下を追加

```ruby
# app/models/reading_session.rb
  # セッション作成後に、作成者をセッション管理者として memberships テーブルに自動的に追加するコールバック
  after_create :assign_creator_as_admin

  # ユーザーをセッションに追加するためのメソッド
  def add_member(user)
    memberships.create!(user: user, session_role: :member, mode: :solo)
  end

  private

  # セッション作成者をセッション管理者として memberships テーブルに追加するメソッド
  def assign_creator_as_admin
    memberships.create(user: creator, session_role: :session_admin, mode: :solo)
  end
```

👉 *`after_create` はレコードが DB に保存された直後に実行され、作成者を `session_admin` として `Membership` に自動登録する。*

## 7️⃣ `ReadingSessions` コントローラーを作成

```bash
touch app/controllers/reading_sessions_controller.rb
```

```ruby
# app/controllers/reading_sessions_controller.rb
class ReadingSessionsController < ApplicationController
  # リーディングセッションへの参加を処理するコントローラー
  before_action :require_login

  # ユーザーが参加しているリーディングセッションの一覧を表示するアクション
  def index
    @memberships = current_user.memberships.includes(reading_session: :text).order(created_at: :desc)
  end

  # 招待トークンを使用してリーディングセッションに参加するアクション
  def join
    session = ReadingSession.find_by(invite_token: params[:token])

    # セッションが見つからない場合は、無効な招待リンクとしてルートにリダイレクト
    if session.nil?
      redirect_to root_path, alert: "無効な招待リンクです"
      return
    end

    # すでにセッションに参加している場合は、テキストのページにリダイレクト
    if session.users.include?(current_user)
      redirect_to text_path(session.text), notice: "すでに参加しています"
      return
    end

    # セッションにユーザーを追加するためのメソッドを呼び出す
    session.add_member(current_user)
    redirect_to text_path(session.text), notice: "セッションに参加しました"
  end
end
```
