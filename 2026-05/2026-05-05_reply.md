```ruby
class Reply < ApplicationRecord
  belongs_to :annotation
  belongs_to :user
  belongs_to :parent, class_name: 'Reply', optional: true
  has_many :children, class_name: 'Reply', foreign_key: :parent_id, dependent: :destroy

  validates :body, presence: true
  validate :parent_belongs_to_same_annotation

  private

  # 同じアノテーションに属しているかを確認するバリデーション
  def parent_belongs_to_same_annotation
    return unless parent_id.present?
    errors.add(:parent_id, :invalid) unless parent&.annotation_id == annotation_id
  end
end
```

```ruby
# app/controllers/replies_controller.rb
class RepliesController < ApplicationController
  before_action :require_login
  before_action :set_annotation
  before_action :require_co_reading, only: :create

  def create
    reply = @annotation.replies.build(reply_params)
    reply.user = current_user
    if reply.save
      render json: {
        id:            reply.id,
        body:          reply.body,
        user_name:     reply.user.name,
        user_id:       reply.user_id,
        parent_id:     reply.parent_id,
        annotation_id: reply.annotation_id
      }, status: :created
    else
      render json: { errors: reply.errors.full_messages }, status: :unprocessable_entity
    end
  end

  def destroy
    reply = @annotation.replies.find(params[:id])
    if reply.user == current_user
      reply.destroy
      render json: {}, status: :ok
    else
      render json: { error: "Not found" }, status: :not_found
    end
  rescue ActiveRecord::RecordNotFound
    render json: { error: "Not found" }, status: :not_found
  end

  private

  def set_annotation
    @annotation = Annotation.where(reading_session: current_user.reading_sessions)
                            .find(params[:annotation_id])
  rescue ActiveRecord::RecordNotFound
    render json: { error: "Not found" }, status: :not_found
  end

  # 共読モードでなければリプライできないようにする
  def require_co_reading
    membership = current_user.memberships.find_by(reading_session: @annotation.reading_session)
    unless membership&.co_reading?
      render json: { error: "共読モードでのみリプライできます" }, status: :forbidden
    end
  end

  def reply_params
    params.require(:reply).permit(:body, :parent_id)
  end
end
```

```js
// app/javascript/controllers/reply_controller.js
import { Controller } from "@hotwired/stimulus"

// targets: Stimulus がテンプレートの data-reply-target 属性を自動でバインドする要素
// values:  ERB から data-reply-*-value 属性で渡す初期値。型を明示することで Stimulus が自動変換する
export default class extends Controller {
  static targets = ["list", "listToggle"]
  static values  = { currentUserId: Number }  // 自分のリプライかどうかを判断するためのログインユーザー ID

  // リプライ一覧の表示・非表示を切り替える
  // toggle は「あれば外す・なければ付ける」なので if/else 不要
  toggleList() {
    this.listTarget.classList.toggle("d-none")
  }

  // 「返信する」ボタンが押されたときに対応するフォームを表示し、parent_id をセットする
  // parent_id が空文字 = トップレベルへの返信、値あり = リプライへの返信
  showForm(event) {
    // stopPropagation で親要素（scrollToAnnotation）へのイベント伝播を止める
    // これがないとフォームを開くと同時にテキスト本文へのスクロールが発火してしまう
    event.stopPropagation()
    const annotationId = event.currentTarget.dataset.annotationId
    const parentId     = event.currentTarget.dataset.parentId || ""

    // data-annotation-id が一致するフォームを探して表示する
    const form = this.element.querySelector(`.reply-form[data-annotation-id="${annotationId}"]`)
    form.querySelector(".reply-parent-id").value = parentId
    form.querySelector(".reply-body").value = ""
    form.classList.remove("d-none")
  }

  cancelForm(event) {
    event.stopPropagation()
    const annotationId = event.currentTarget.dataset.annotationId
    const form = this.element.querySelector(`.reply-form[data-annotation-id="${annotationId}"]`)
    form.classList.add("d-none")
  }

  // async/await: fetch はサーバーとの通信（非同期処理）なので、完了を待つために await を使う
  // await がないとレスポンスを受け取る前に次の行が実行されてしまう
  async submitReply(event) {
    event.stopPropagation()
    const annotationId = event.currentTarget.dataset.annotationId
    const form         = this.element.querySelector(`.reply-form[data-annotation-id="${annotationId}"]`)
    const body         = form.querySelector(".reply-body").value.trim()
    const parentId     = form.querySelector(".reply-parent-id").value || null

    if (!body) return

    const response = await fetch(`/annotations/${annotationId}/replies`, {
      method:  "POST",
      headers: {
        "Content-Type": "application/json",
        // Rails の CSRF 保護：サーバーはこのトークンで「正規のフォームからのリクエスト」かを検証する
        // コメントやアノテーションの送信と同じパターン
        "X-CSRF-Token": this.#csrfToken()
      },
      body: JSON.stringify({ reply: { body, parent_id: parentId } })
    })

    if (!response.ok) return

    // レスポンス JSON からリプライデータを取り出して DOM に追加する
    const reply = await response.json()
    this.#insertReply(reply)
    form.classList.add("d-none")
  }

  async deleteReply(event) {
    event.stopPropagation()
    const replyId      = event.currentTarget.dataset.replyId
    const annotationId = event.currentTarget.dataset.annotationId

    const response = await fetch(`/annotations/${annotationId}/replies/${replyId}`, {
      method:  "DELETE",
      headers: { "X-CSRF-Token": this.#csrfToken() }
    })

    if (!response.ok) return

    // data-reply-id が一致する要素を DOM から取り除く
    // ?. はオプショナルチェーン：要素が見つからない場合でも例外にならず undefined を返す
    this.element.querySelector(`.reply-item[data-reply-id="${replyId}"]`)?.remove()
  }

  // ── プライベートメソッド（# プレフィックス）──────────────────────────────
  // クラス外からは呼び出せない内部処理。Stimulus の action 属性からも呼べない。

  // サーバーからの JSON を見て DOM の適切な場所にリプライを挿入する
  // parent_id なし → アノテーションの reply-list 末尾
  // parent_id あり → 親リプライの reply-children 末尾（インデントを +16px）
  #insertReply(reply) {
    const html = this.#buildReplyHtml(reply)

    if (reply.parent_id) {
      const parentItem     = this.element.querySelector(`.reply-item[data-reply-id="${reply.parent_id}"]`)
      const childContainer = parentItem?.querySelector(".reply-children")
      // 親の margin-left を読み取って子のインデントを算出する
      const parentMargin   = parseInt(parentItem?.style.marginLeft || "0", 10)
      const childHtml      = html.replace(`margin-left: 0px`, `margin-left: ${parentMargin + 16}px`)
      childContainer?.insertAdjacentHTML("beforeend", childHtml)
    } else {
      const list = this.element.querySelector(`.reply-list[data-annotation-id="${reply.annotation_id}"]`)
      // insertAdjacentHTML("beforeend", ...) は要素の末尾の子として HTML を追加する
      list?.insertAdjacentHTML("beforeend", html)
    }
  }

  // DOM に挿入するリプライの HTML 文字列を組み立てる
  // ERB の _reply.html.erb と同じ構造にすることで、初期描画と動的追加の見た目を一致させる
  #buildReplyHtml(reply) {
    // 自分のリプライにだけ削除ボタンを表示する（サーバー側と同じ判定）
    const isOwn     = reply.user_id === this.currentUserIdValue
    const deleteBtn = isOwn
      ? `<button class="btn btn-link btn-sm p-0 ms-auto text-danger"
                 data-action="click->reply#deleteReply"
                 data-reply-id="${reply.id}"
                 data-annotation-id="${reply.annotation_id}"
                 style="font-size: 0.75rem;">削除</button>`
      : ""
    // toLocaleString でブラウザのロケールに合わせた日時フォーマットを生成する
    const now = new Date().toLocaleString("ja-JP", {
      year: "numeric", month: "2-digit", day: "2-digit",
      hour: "2-digit", minute: "2-digit"
    })

    // テンプレートリテラル（バッククォート）で複数行の HTML 文字列を作る
    // margin-left: 0px は #insertReply 側で親のインデントに応じて上書きされる
    return `
      <div class="reply-item" style="margin-left: 0px;" data-reply-id="${reply.id}">
        <div class="d-flex align-items-center gap-2 mb-1">
          <strong class="small">${reply.user_name}</strong>
          <span class="text-muted" style="font-size: 0.75rem;">${now}</span>
          ${deleteBtn}
        </div>
        <p class="mb-1 small">${reply.body}</p>
        <button class="btn btn-link btn-sm p-0"
                data-action="click->reply#showForm"
                data-annotation-id="${reply.annotation_id}"
                data-parent-id="${reply.id}"
                style="font-size: 0.75rem;">返信する</button>
        <div class="reply-children"></div>
      </div>`
  }

  // meta タグから CSRF トークンを取得する
  // Rails はページ読み込み時に <meta name="csrf-token"> をヘッダーに埋め込んでいる
  #csrfToken() {
    return document.querySelector('meta[name="csrf-token"]')?.content
  }
}
```
