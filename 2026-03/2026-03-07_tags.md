## 1️⃣ Tag モデルを生成

```bash
$ docker compose exec web rails g model Tag user:references name:string
      invoke  active_record
      create    db/migrate/20260307024210_create_tags.rb
      create    app/models/tag.rb
      invoke    test_unit
      create      test/models/tag_test.rb
      create      test/fixtures/tags.yml
```

⬇️ 関連ファイル を修正

```ruby
# g コマンドで自動生成されたマイグレーションファイル
class CreateTags < ActiveRecord::Migration[7.2]
  def change
    create_table :tags do |t|
      t.references :user, null: false, foreign_key: true
      t.string :name, null: false # ⬅️タグ名は必須

      t.timestamps
    end

    # ⬇️ 同一ユーザーは同一タグ名をつくれない（他のユーザーは可）
    add_index :tags, [:user_id, :name], unique: true
  end
end
```

```ruby
# app/models/tag.rb
class Tag < ApplicationRecord
  belongs_to :user

  # ⬇️ 多対多
  has_many :record_tags, dependent: :destroy
  has_many :learning_records, through: :record_tags
end
```

⬇️ マイグレーション

## 2️⃣ RecordTag モデルを生成

```bash
$ docker compose exec web rails g model RecordTag learning_record:references tag:references
      invoke  active_record
      create    db/migrate/20260307031616_create_record_tags.rb
      create    app/models/record_tag.rb
      invoke    test_unit
      create      test/models/record_tag_test.rb
      create      test/fixtures/record_tags.yml
```

⬇️ 自動生成された migrate ファイルを修正

```ruby
class CreateRecordTags < ActiveRecord::Migration[7.2]
  def change
    create_table :record_tags do |t|
      t.references :learning_record, null: false, foreign_key: true
      t.references :tag, null: false, foreign_key: true

      t.timestamps
    end

    # 同一タブを同一レコードに重複しないようにする制約
    add_index :record_tags,
              [:learning_record_id, :tag_id],
              unique: true
  end
end
```

⚠️ *ここで `db:migrate` を行うと参照テーブルが未実装のためエラーになるので、実装後にまとめて実行する*

```ruby
class CreateTodoTags < ActiveRecord::Migration[7.2]
  def change
    create_table :todo_tags do |t|
```

## CI の自動テスト対応

g コマンドでテストファイルをつくらないようにする。

```ruby
# config/application.rb
  class Application < Rails::Application
    # ... 既存設定 ...

    # generator設定
    config.generators do |g|
      g.test_framework nil  # テストファイルを作らない
    end
  end
```

👉 *毎回 `--skip-test` を書かないで済むようになる*
