```ruby
# app/controllers/todos_controller.rb
class TodosController < ApplicationController
  before_action :set_learning_theme
  before_action :set_todo, only: %i[edit update destroy]

  def index
    @todos = @learning_theme.todos.order(created_at: :desc)
  end

  def new
    @todo = @learning_theme.todos.build
  end

  def create
    @todo = @learning_theme.todos.build(todo_params)
    @todo.user = current_user

    if @todo.save
      redirect_to learning_theme_todos_path(@learning_theme), notice: "Todoを保存しました"
    else
      flash.now[:alert] = "Todoの保存に失敗しました"
      render :new, status: :unprocessable_entity
    end
  end

  def edit
  end

  def update
    if @todo.update(todo_params)
      redirect_to learning_theme_todos_path(@learning_theme), notice: "Todoを更新しました"
    else
      flash.now[:alert] = "Todoの更新に失敗しました"
      render :edit, status: :unprocessable_entity
    end
  end

  def destroy
    @todo.destroy
    redirect_to learning_theme_todos_path(@learning_theme), notice: "Todoを削除しました"
  end

  private

  def todo_params
    params.require(:todo).permit(:title, :description, :is_completed)
  end

  def set_learning_theme
    # current_user の learning_themes の中からのみ検索することで、他ユーザーの learning_theme にアクセスできないようにする
    @learning_theme = current_user.learning_themes.find(params[:learning_theme_id])
  end

  def set_todo
    # @learning_theme の todos の中からのみ検索することで、他ユーザーの todo にアクセスできないようにする
    @todo = @learning_theme.todos.find(params[:id])
  end
end
```

- `set_learning_theme` を全アクションで実行 *（ネストしているため）*
- `create` で `@todo.user = current_user` を明示的にセット *（`learning_theme` 経由で build するため `user_id` が自動でセットされない）*

- Todo 一覧 `app/views/todos/index.html.erb`

  👉 *`Todo` にはシンプルなトグルボタンを置いて「完了 / 未完了」を視覚的に区別して表示させる*

```erb
<div class="container mt-5" style="max-width: 600px;">
  <h1 class="h4"> TODO リスト</h1>

  <%= link_to '新規 TODO を追加', new_learning_theme_todo_path(@learning_theme), class: "btn btn-sm btn-primary" %>
  <%= link_to '学習ログ', learning_records_path, class: "btn btn-sm btn-outline-primary" %>

  <table class="table table-hover">
    <thead>
      <tr>
        <th>My TODOs</th>
      </tr>
    </thead>
    <tbody>
      <% @todos.each do |todo| %>
        <tr>
          <td class="<%= 'text-decoration-line-through text-muted' if todo.is_completed %>">
            <%= todo.title %>
          </td>
          <td>
            <%= button_to todo.is_completed ? '取り消し' : '完了にする',
                learning_theme_todo_path(@learning_theme, todo),
                method: :patch,
                params: { todo: { is_completed: !todo.is_completed } },
                class: "btn btn-sm #{todo.is_completed ? 'btn-outline-secondary' : 'btn-outline-success'}" %>
          </td>
          <td>
            <%= link_to '編集', edit_learning_theme_todo_path(@learning_theme, todo), class: "btn btn-sm btn-outline-secondary" %>
            <%= link_to '削除', learning_theme_todo_path(@learning_theme, todo), class: "btn btn-sm btn-outline-danger", data: { turbo_method: :delete, turbo_confirm: '本当に削除しますか？' } %>
          </td>
        </tr>
      <% end %>
    </tbody>
  </table>

  <p class="text-muted small mt-3">※ 今後の学習計画や参照資料などを TODO リスト に追加して管理できます</p>
</div>
```

  📝 トグルボタン

  | コード | 動作 |
  | --- | --- |
  | `button_to '完了にする', learning_theme_todo_path(...)` | 指定したパスにリクエストを送るボタンを生成 |
  | `method: :patch` | `update` アクションを呼び出すために `PATCH` メソッドを指定 |
  | `params: { todo: { is_completed: !todo.is_completed } }` | `!todo.is_completed` で値を反転し、コントローラーの `update` アクションがこれを受け取って保存 |
  | `todo.is_completed ? '未完了にする' : '完了にする'` | 状態によってラベルとCSSを切り替える |
